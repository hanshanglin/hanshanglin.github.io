class Solution {
public:
    int countNegatives(vector<vector<int>>& grid) {
        int ans = 0;
        for(auto i : grid)
            for(auto j:i)
                if(j<0)ans++;
        return ans;
    }
};

class ProductOfNumbers {
public:
    vector<int> list;
    ProductOfNumbers() {}

    void add(int num) {
        list.push_back(num);
    }
    
    int getProduct(int k) {
        int ans=1;
        for(int i = 0;i<k;i++)
            ans*=list[list.size()-1-i];
        return ans;
    }
};

class Solution {
public:
    int maxEvents(vector<vector<int>>& events) {
        multiset<pair<int,int>> tmp;
        unordered_map<int,bool> visited;
        for(int i = 0 ;i<events.size();i++)
            tmp.insert({events[i][1],events[i][0]});
        int ans = 0;
        for(auto i=tmp.begin();i!=tmp.end();i++){
            int start = i->second;
            int end = i->first;
            for(int day = start;day<end+1;day++){
                if(visited.find(day) == visited.end()){
                    visited[day]=true;
                    ans++;
                    break;
                }
            }
        }
        return ans; 
    }
};

class Solution {
public:
    bool isPossible(vector<int>& target) {
        priority_queue<long long> pq(target.begin(),target.end());
        long long s = accumulate(target.begin(),target.end(),0LL);
        long long t = 0, remin = 0;
        while (pq.top()!=1){
            t = pq.top();pq.pop();
            remin = s - t;
            if ((2*t - s) < 1) return false;
            pq.push((2*t - s));
            s = t;
        }
        return true;
    }
};
